YAML 語法範例
asset_definition: metadata: name: "Compressor_GA75" version: "1.0" category: "MEP/Compressed_Air" data_tags: - name: "pressure" type: "float" unit: "bar" protocol: "OPC_UA" address: "ns=2;s=Compressor.Pressure" - name: "status" type: "string" enum: ["Running", "Idle", "Fault"] geometry: model_file: "compressor_ga75.usd" dimensions: width: 1.2 # meters depth: 0.8 height: 1.5 behavior: script: "compressor_control.py" events: - trigger: "pressure > 8.0" action: "shutdown"
